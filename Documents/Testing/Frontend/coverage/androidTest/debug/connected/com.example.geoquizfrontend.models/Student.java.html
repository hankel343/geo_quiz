package com.example.geoquizfrontend.models;

public class Student {

    private String lastName;
    private String email;
    private String password;
    private String classCode;
    private float grade;

    public Student() {

    }

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

        return firstName;

        this.firstName = name;

}
